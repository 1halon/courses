\begin{problem}[$ \frac{a + 3b + 2c}{2a - b + c} = 2 $ ise $ \frac{2a + 3b}{2b - a} = ? $]
\end{problem}

\begin{proof}[\textit{ Sol. }]
  \begin{equation*}
    \begin{aligned}[t]
      &a + 3b + 2c = 4a - 2b + 2c\\
      &5b = 3a, a = 5k, b = 3k
    \end{aligned}
    \qquad\qquad
    \begin{aligned}[t]
      \frac{10k + 9k}{6k - 5k} = \frac{19k}{k} = 19
    \end{aligned}
  \end{equation*}
\end{proof}
